ivoid bubblesort(string * a, int n) {
	bool swapped = true;
	for (int i = 0; i < n - 1 && swapped; i++) {
		swapped = false;
		for (int j = n - 1; j > i; j--) {
			if (a[j] <= a[j - 1]) {
				swap(a[j], a[j - 1]);
				swapped = true;
			}
		}
	}

}
